## Install

```bash
npm install --save orsys-reactive-store
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'orsys-reactive-store'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
